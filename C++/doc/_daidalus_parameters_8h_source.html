1056template <typename T>T parameter_data(const ParameterData& p, const std::string& key, Function<const std::string&,T> &f){
1057 T default_val = f.apply(key);
1058 if (p.contains(key)) {
1059 return default_val;
1060 }
1061 DaidalusParameters::aliasestype::const_iterator find_ptr = DaidalusParameters::getAliases().find(key);
1062 if (find_ptr == DaidalusParameters::getAliases().end()) {
1063 return default_val;
1064 }
1065 std::vector<std::string>::const_iterator alias_ptr;
1066 for (alias_ptr = find_ptr->second.begin(); alias_ptr != find_ptr->second.end(); ++alias_ptr) {
1067 if (p.contains(*alias_ptr)) {
1068 return f.apply(*alias_ptr);
1069 }
1070 }
1071 return default_val;
1072}
